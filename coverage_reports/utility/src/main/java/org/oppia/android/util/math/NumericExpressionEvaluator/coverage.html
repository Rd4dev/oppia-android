36 /**
37 * Evaluates a math expression.
38 *
39 * This function only works with numeric expressions since variable expressions have no means
40 * for evaluation (so they'll always result in a ``null`` return value).
41 *
42 * The function generally attempts to retain the most precise representation of a value in the
43 * following order (from highest priority to lowest):
44 * 1. Integers
45 * 2. Fractions (rational values)
46 * 3. Doubles (irrational values)
47 *
48 * Doubles will only be used if there's no other choice as they do not have perfect precision
49 * unlike the other two structures. Further, it's possible for doubles to be used in cases where
50 * an integer could work, or fractions to represent whole integers (due to quirks in underlying
51 * routines). That being said, within a certain precision threshold values returned by this
52 * function should be deterministic across multiple calls (for the same [MathExpression]).
53 *
54 * There are a number of cases where this function will fail:
55 * - When trying to evaluate a variable expression.
56 * - When trying to evaluate an invalid [MathExpression] (i.e. one of the substructures within
57 * the expression is not actually initialized per the proto structures).
58 * - When trying to perform an impossible math operation (such as divide by zero). Note that
59 * this will sometimes result in a [Real] being returned with a value like NaN or infinity,
60 * and other times may result in an exception being thrown.
61 *
62 * Note that there's no guard against overflowing values during computation, so care should be
63 * taken by the caller that this is possible for certain expressions.
64 *
65 * For more specifics on the constituent operations that "power" this function, see:
66 * - [Real.plus]
67 * - [Real.minus]
68 * - [Real.times]
69 * - [Real.div]
70 * - [Real.pow]
71 * - [Real.unaryMinus]
72 * - [sqrt]
73 *
74 * @return the [Real] representing the evaluated expression, or ``null`` if something went wrong
75 */
76 fun MathExpression.evaluate(): Real? {
77 return when (expressionTypeCase) {
78 CONSTANT -> constant
79 VARIABLE -> null // Variables not supported in numeric expressions.
80 BINARY_OPERATION -> binaryOperation.evaluate()
81 UNARY_OPERATION -> unaryOperation.evaluate()
82 FUNCTION_CALL -> functionCall.evaluate()
83 GROUP -> group.evaluate()
84 EXPRESSIONTYPE_NOT_SET, null -> null
85 }
86 }

88 private fun MathBinaryOperation.evaluate(): Real? {
89 return when (operator) {
90 ADD -> rightOperand.evaluate()?.let { leftOperand.evaluate()?.plus(it) }
91 SUBTRACT -> rightOperand.evaluate()?.let { leftOperand.evaluate()?.minus(it) }
92 MULTIPLY -> rightOperand.evaluate()?.let { leftOperand.evaluate()?.times(it) }
93 DIVIDE -> rightOperand.evaluate()?.let { leftOperand.evaluate()?.div(it) }
94 EXPONENTIATE -> rightOperand.evaluate()?.let { leftOperand.evaluate()?.pow(it) }
95 BinaryOperator.OPERATOR_UNSPECIFIED, BinaryOperator.UNRECOGNIZED, null -> null
96 }
97 }

99 private fun MathUnaryOperation.evaluate(): Real? {
100 return when (operator) {
101 NEGATE -> operand.evaluate()?.let { -it }
102 POSITIVE -> operand.evaluate() // '+2' is the same as just '2'.
103 UnaryOperator.OPERATOR_UNSPECIFIED, UnaryOperator.UNRECOGNIZED, null -> null
104 }
105 }
